
``` rust

use std::net::UdpSocket;
use std::io::Error;

fn main() -> Result<(), Error> {
    let socket = UdpSocket::bind("127.0.0.1:0")?;
    let target = "127.0.0.1:34254";

    let number: i32 = 42;
    let bytes = number.to_be_bytes(); // Convert the integer to bytes

    socket.send_to(&bytes, target)?;

    println!("Sent: {}", number);
    Ok(())
}
```
